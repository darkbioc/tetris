/**

 * @author victor
 */
public class ScoreBoard extends JLabel{
    
    private int score;
    
    public ScoreBoard() {
        super();
        score = 0;
        setText("Score: "+score);
    }
    
    public void increment(int points) {
        score += points;
        setText("Score: "+score);
    }
    
    public void reset() {
        score = 0;
        setText("Score: "+score);
    }
    
    public void pause()
    {
        setText("PAUSED");
    }
    
    public void resume()
    {
        setText("Score: "+score);
    }
    
}
